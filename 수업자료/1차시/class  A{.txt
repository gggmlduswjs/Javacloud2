class  A{
       
     public int  getIndex( String[] list , String target){
              int index=-1;
              for( int i=0; i < list.length; i++){ 
                   String item  = list[i];
                   if( item.equals( target){
                           index = i;        
                           break;                   
                    }
              }
            return index;
     }
}


class AMain{

               public static void main( String[] args){
                  String[] strings = {"apple", "banana", "orange", "grape", "kiwi"};   
                  int result= getIndex( strings, "orange");
               }
}
